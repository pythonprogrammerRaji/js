By object literal

Syntax :

        let literalObjectName = {
            "property1":"value1",
            "property2":"value2",
            "property3":"value3",
            "propertyN":"valueN",
           
        };

        console.log(literalObjectName);
        console.log(typeof(literalObjectName));
        console.log(literalObjectName.property1);
